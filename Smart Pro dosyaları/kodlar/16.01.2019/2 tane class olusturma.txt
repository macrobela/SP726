using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace NestedClass
{
    class KisiselBilgi
    {
        public string Ad;
        public string Soyad;

        public YasBilgi _yasBilgisi;

        public KisiselBilgi()
        {
            _yasBilgisi = new YasBilgi();
        }

        // Yukarýdaki nesne tanýmlama sayesinde
        // YasBilgisi sýnýfýnýn üyelerine bu sýnýf nesnesi aracýlýðý ile deðer atayabiliriz.
        // Metotlarýný kullanýlabiliriz.
        public void Yazdir()
        {
            Console.WriteLine(Ad+"   "+Soyad);
            Console.WriteLine(_yasBilgisi.Yas);
        }
    }
    class YasBilgi
    {
        public int Yas;
        public void YasYazdir()
        {
            Console.WriteLine(Yas);   
        }
    }
    class Program
    {
        static void Main(string[] args)
        {
            KisiselBilgi kb1 = new KisiselBilgi();
            kb1.Ad = "SmartPro";
            kb1.Soyad = "Kadýköy";
            kb1._yasBilgisi.Yas = 55;
            kb1.Yazdir();
            Console.WriteLine(kb1._yasBilgisi.Yas);
            Console.ReadKey();
        }
    }
}
